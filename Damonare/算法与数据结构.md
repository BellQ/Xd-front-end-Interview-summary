1. 输入两个链表，分别代表两个数字的倒序排列如下：
342+465 = 807;

> Input: (2 -> 4 -> 3) + (5 -> 6 -> 4)
Output: 7 -> 0 -> 8

```javascript
/**
 * ListNode类用来生成具体的节点
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} l1
 * @param {ListNode} l2
 * @return {ListNode}
 */
var addTwoNumbers = function(l1, l2) {
    var l = new ListNode(0);
    var head = l;
    var sum = 0;
    var front = 0;
    while (l1 !== null || l2 !== null || sum === 1) {
        if (l1) {
            sum = sum + l1.val;
            l1 = l1.next;
        }
        if (l2) {
            sum = sum + l2.val;
            l2 = l2.next;
        }
        if (sum >= 10) {
            front = 1;
            sum = sum - 10;
        }
        head.next = new ListNode(sum);
        head = head.next;

        sum = front;
        front = 0;
    }
    return l.next;
};
```
